原生Js封装Ajax使用方法

2018-12-06

日期: 2018-12-06 20:10:14

为了更好理解jQuery的$.ajax的方法，使用原生js封装了一个仿jQuery的函数。

从传入单个优化为传入整个对象，再利用对象的属性获取属性值

1.处理传入文件的路径

/**
 * 处理文件路径，传递参数
 * @param data {object}
 * @returns {string}
 */
function objToString(data) {
   /* data.t = new Date().getTime();*/
var res = [];
for (var key in data) {
res.push(encodeURIComponent(key) + '=' + encodeURIComponent(data[key]));//遇到中文时编码
}
return res.join('&');
}

2.封装ajax请求

function Ajax(option) {
//0.处理url
var str = objToString(option.data);

//1.创建ajax对象//兼容写法
var xhr, timer;
if (window.XMLHttpRequest) {// code for IE7+, Firefox, Chrome, Opera, Safari
    xhr = new XMLHttpRequest();
}
else {// code for IE6, IE5
    xhr = new ActiveXObject("Microsoft.XMLHTTP");
}

//2.判断传输类型
if (option.type.toLowerCase() === 'get') {//统一转换为小写判断
    xhr.open(option.type, option.url + '?' + str, true);
    xhr.send();
} else {
    xhr.open(option.type, option.url, true);
    xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");//POST模式下需要添加请求头
    xhr.send(str);//发送需要的数据
}

//3.处理响应事件
xhr.onreadystatechange = function () {
    if (xhr.readyState === 4) {
        clearInterval(timer);//已响应则关掉超时操作
        if (xhr.status >= 200 && xhr.status < 300 || xhr.status === 304) {
            option.success(xhr);//成功，回调函数
        } else {
            option.error(xhr);//失败，回调函数
            console.log(str);
        }
    }
};

//4.设置超时操作
if (option.timeout) {
    timer = setInterval(function () {
        alert('请求超时！');
        xhr.abort();
        clearInterval(timer);
    }, option.timeout);
}
}
