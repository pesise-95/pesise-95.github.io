原生Js封装使用Cookie使用方法

2018-12-06

日期: 2018-12-06 20:23:16

cookie的一些注意点

//1.cookie是浏览器使用的一种缓存数据，大小一般为4k，条数为20-50条。

//1.1 创建cookie的最基本模式，cookie只能保存一个key=value的值；

document.cookie = "name=张三";

//1.2 创建时可选参数expires为过期时间，默认为一次会话，path为保存路径，默认为当前打开地址的目录，即关闭浏览器cookie自动删除

document.cookie = "age=18; expires=Thu, 18 Dec 2018 12:00:00 GMT; path=/";

//1.3 同浏览器下保存一条相同的cookie，无论打开哪一个网页都可以查看到。

//1.4 封装一个创建cookie的函数

/**
    * 创建cookie
    * @param key {string}
    * @param value {string}
    * @param exdays {number} [距离当前时间的天数]
    * @param path {string} [保存路径]
    * @param domain [解析域名]
    */
   function addCookie(key, value, exdays, path, domain) {
       //1.处理保存路径
       let index = document.location.pathname.lastIndexOf("/");
       let currentPath = document.location.pathname.slice(index);
       path = path || currentPath;

       //2.处理domain
       domain = domain || document.domain;

       //3.处理过期时间
       if (!exdays) {
           document.cookie = key + "=" + value + "; +path=" + path + ";domain" + domain + ";"
       } else {
           let date = new Date();
           date.setTime(date.getTime() + (exdays * 24 * 60 * 60 * 1000));
           let expires = "expires=" + date.toGMTString();
           document.cookie = key + "=" + value + "; " + expires;
       }
   }

//2.封装一个获取cookie的函数

/**
    * 获取cookie
    * @param key {string}
    * @returns {string}
    */
   function getCookie(key) {
       let res = document.cookie.split(';');
       for (var i = 0; i < res.length; i++) {
           let temp = res[i].split('=');
           if (temp[0].trim() === key) {
               return temp[1];
           }
       }
   }

//3.删除cookie

function delCookie(key) {
        addCookie(key,getCookie(key),-1);//时间为-1则代表立即删除
    }
