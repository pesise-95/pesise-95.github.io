title:ES6的Class语法糖+Canvas绘制炫丽小球

日期: 2018-12-30 22:36:10

最近工作比较忙，很少来更新。恰逢元旦假期，复习一下ES6的语法，结合H5的绘图制作一个炫丽小球。效果如下：

Js代码如下，主要是灵活运用函数的继承，使用到了underscore的类库

window.onload = function () {
    //1.添加画布
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    canvas.width = 1900;
    canvas.height = 920;
    canvas.style.backgroundColor = '#000';

    //2.增加小球类
    class Ball {
        //2.1 创建小球状态
        constructor(x, y, color) {
            this.x = x;
            this.y = y;
            this.color = color;
            this.r = 40;
        }

        //2.2 创建绘制小球的方法
        pain() {
            ctx.save();
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.r, 0, Math.PI * 2);
            ctx.fillStyle = this.color;
            ctx.fill();
            ctx.restore();
        }
    }

    //3.创建会移动的小球类
    //3.1 继承小球类
    class MoveBall extends Ball {
        //3.1.1 继承父类的属性
        constructor(x, y, color) {
            super(x, y, color);
            //3.1.2 设置增量
            this.dX = _.random(-5, 5);
            this.dY = _.random(-5, 5);
            this.dR = _.random(1, 3);
        }

        //3.1.3 设置增量方法
        upDate() {
            this.x += this.dX;
            this.y += this.dY;
            this.r -= this.dR;
            if (this.r < 0) {
                this.r = 0
            }
        }
    }

    //4. 实例化小球
    let ballArr = [];
    let colorArr = ['red', 'blue', 'green', 'orange', 'pink', 'white', 'yellow'];

    //5. 鼠标移动绘图
    canvas.addEventListener('mousemove', function (e) {
        //5.1 产生对应位置和随机颜色的小球
        ballArr.push(new MoveBall(e.clientX, e.clientY, colorArr[_.random(0, colorArr.length - 1)]))
    });

    //6.开启定时器
    setInterval(function () {
        //6.1 清除画布
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        //6.2 绘制开始
        for (let i = 0; i < ballArr.length; i++) {
            ballArr[i].pain();
            ballArr[i].upDate();
        }
    }, 50);
}
